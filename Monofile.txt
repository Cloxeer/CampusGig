// ── UTILS ─────────────────────────────────────────────────────────────────────
function useTimer() {
  const [t,setT] = useState(0);
  useEffect(() => { const id=setInterval(()=>setT(x=>x+1),1000); return ()=>clearInterval(id); },[]);
  return t;
}
function elapsed(ts) {
  const s=Math.floor((Date.now()-ts)/1000);
  if(s<60) return `${s}s ago`;
  const m=Math.floor(s/60); if(m<60) return `${m}m ago`;
  const h=Math.floor(m/60); if(h<24) return `${h}h ago`;
  return `${Math.floor(h/24)}d ago`;
}

      {showMap&&(
        <div className="overlay" onClick={()=>setShowMap(false)}>
          <div className="sheet slidein" onClick={e=>e.stopPropagation()}>
            <div className="handle"/>
            <div style={{padding:"16px 20px 14px",borderBottom:"1px solid var(--bd)",display:"flex",alignItems:"center",justifyContent:"space-between"}}>
              <div>
                <div style={{fontSize:16,fontWeight:600,letterSpacing:"-.02em"}}>Gigs near you</div>
                <div style={{fontSize:12,color:"var(--fg3)",fontFamily:"var(--mono)",marginTop:2}}>{filteredGigs.length} open · UC Davis campus</div>
              </div>
              <button className="btn bg-btn bico" onClick={()=>setShowMap(false)}><X size={15}/></button>
            </div>
            <div style={{padding:"16px"}}>
              <div className="map-container">
                <CampusMap/>
              </div>
              <div style={{marginTop:12,fontSize:12,color:"var(--fg3)",display:"flex",alignItems:"center",gap:8}}>
                <div style={{width:10,height:10,borderRadius:"50%",background:"var(--green)",flexShrink:0}}/>
                <span>Your location &nbsp;·&nbsp; Colored dots = open gigs</span>
              </div>
            </div>
            <div style={{padding:"0 16px",display:"flex",flexDirection:"column",gap:6}}>
              {filteredGigs.slice(0,3).map(g=>(
                <div key={g.id} style={{display:"flex",alignItems:"center",gap:10,padding:"10px 12px",background:"var(--bg3)",borderRadius:"var(--r)",border:"1px solid var(--bd)",cursor:"pointer"}}
                  onClick={()=>{setShowMap(false);setSelectedGig(g);setRequested(false);}}>
                  <div style={{width:8,height:8,borderRadius:"50%",background:g.color,flexShrink:0}}/>
                  <div style={{flex:1,minWidth:0}}>
                    <div style={{fontSize:13,fontWeight:500,color:"var(--fg)",overflow:"hidden",textOverflow:"ellipsis",whiteSpace:"nowrap"}}>{g.title}</div>
                    <div style={{fontSize:11,color:"var(--fg3)",fontFamily:"var(--mono)"}}>{g.loc} · {g.eta}</div>
                  </div>
                  <span style={{fontSize:14,fontWeight:700,fontFamily:"var(--mono)",color:"var(--fg)",flexShrink:0}}>{g.price}</span>
                </div>
              ))}
            </div>
            <div style={{height:24}}/>
          </div>
        </div>
      )}
    </div>
  );
}
